Konfigurasi Web Server Apache

Langkah-langkah dasar untuk mengkonfigurasi Apache di Debian.

# 1. Instalasi Apache
sudo apt update
sudo apt install apache2

# 2. Memulai dan mengaktifkan service
sudo systemctl start apache2
sudo systemctl enable apache2

# 3. Konfigurasi virtual host (contoh)
sudo nano /etc/apache2/sites-available/contoh.com.conf

# Isi file virtual host:

    ServerName contoh.com
    ServerAlias www.contoh.com
    DocumentRoot /var/www/contoh.com
    ErrorLog ${APACHE_LOG_DIR}/contoh.com-error.log
    CustomLog ${APACHE_LOG_DIR}/contoh.com-access.log combined


# 4. Aktifkan virtual host
sudo a2ensite contoh.com.conf
sudo systemctl reload apache2

Konfigurasi DNS Server (Bind9)

Langkah-langkah mengkonfigurasi DNS menggunakan Bind9.

# 1. Instalasi Bind9
sudo apt update
sudo apt install bind9 dnsutils

# 2. Konfigurasi zona
sudo nano /etc/bind/named.conf.local

# Tambahkan:
zone "contoh.com" {
    type master;
    file "/etc/bind/db.contoh.com";
};

# 3. Buat file zona
sudo cp /etc/bind/db.local /etc/bind/db.contoh.com
sudo nano /etc/bind/db.contoh.com

# Contoh isi:
$TTL    604800
@       IN      SOA     ns1.contoh.com. admin.contoh.com. (
                              2         ; Serial
                         604800         ; Refresh
                          86400         ; Retry
                        2419200         ; Expire
                         604800 )       ; Negative Cache TTL
@       IN      NS      ns1.contoh.com.
@       IN      A       192.168.0.2
ns1     IN      A       192.168.0.2
www     IN      A       192.168.0.2

# 4. Restart service
sudo systemctl restart bind9
